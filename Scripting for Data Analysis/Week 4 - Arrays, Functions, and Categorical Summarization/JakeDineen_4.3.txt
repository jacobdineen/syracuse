def count_numbers():
    summation = 0
    count = 0
    average = 0

    while True:
        try:
            x = input("Enter a number: ")
            if (x == "done"):
                break
            value = float(x)
            summation += value 
            count +=1
            average = summation / count
        except:
            print("Invalid input.")
    print('Count: {} | Sum: {} | Avg: {}'.format(count,summation,average))
	
count_numbers()
